    F - Girlfriend

    算法 > 题解 2021牛客暑期多校第二场
    发布时间 : 2021-07-20 09:53
    阅读 :
    1. 题目描述
    2. 解题思路
    3. 代码

    题目描述

    给出四个空间点\(A,B,C,D\)和\(k_{1},k_{2}\)，求关于动点\(P_{1},P_{2}\)且满足\(|AP_{1}| \geq k_{1}|BP_{1}|, |CP_{2}| \geq k_{2}|DP_{2}|\)的两个点的轨迹方程\(\Gamma_{P1},\Gamma_{P2}\)，并求出它们的交体积。

    解题思路

    首先进行轨迹方程的推导。设\(A(x_{1}, y_{1}, z_{1}), B(x_{2}, y_{2}, z_{2})\)，由题目条件得： \[ (x-x_{1})^2+(y-y_{1})^2+(z-z_{1})^2 \geq k^2((x-x_{2})^2+(y-y_{2})^2+(z-z_{2})^2) \] 配方后，化简得： \[ (x-\frac{k^2 x_{2}-x_{1}}{k^2-1})^2 + (y-\frac{k^2 y_{2}-y_{1}}{k^2-1})^2 + (z-\frac{k^2 z_{2}-z_{1}}{k^2-1})^2 \leq (\frac{k^2 x_{2}-x_{1}}{k^2-1})^2-\frac{x_{2}^2k^2-x_{1}}{k^2-1} + (\frac{k^2 y_{2}-y_{1}}{k^2-1})^2-\frac{y_{2}^2k^2-y_{1}}{k^2-1} + (\frac{k^2 z_{2}-z_{1}}{k^2-1})^2-\frac{z_{2}^2k^2-z_{1}}{k^2-1} \] 得到圆心： \[ (\frac{k^2 x_{2}-x_{1}}{k^2-1},\frac{k^2 y_{2}-y_{1}}{k^2-1},\frac{k^2 z_{2}-z_{1}}{k^2-1}) \] ，半径： \[ \sqrt{(\frac{k^2 x_{2}-x_{1}}{k^2-1})^2-\frac{x_{2}^2k^2-x_{1}}{k^2-1} + (\frac{k^2 y_{2}-y_{1}}{k^2-1})^2-\frac{y_{2}^2k^2-y_{1}}{k^2-1} + (\frac{k^2 z_{2}-z_{1}}{k^2-1})^2-\frac{z_{2}^2k^2-z_{1}}{k^2-1}} \] 然后算两个球的交体积。用微积分分别算出两个部分的体积，下面这个积分是绿色区域的体积：

    \[ \int_{0}^h\pi(R^2-(Rcos\alpha+x)^2)dx = \pi (Rh^2-\frac{h^3}{3}) \]

    圆的相交体积大小的结论为，其中d为球心距： \[ cos\alpha = \frac{R^2+d^2-r^2}{2Rd}\\cos\beta = \frac{r^2+d^2-R^2}{2rd}\\V=\frac{\pi}{3}((3R-R(1-cos\alpha))R^2(1-cos\alpha)^2+(3r-r(1-cos\beta))r^2(1-cos\beta)^2) \]

    代码

    #include <cstdio>
#include <cmath>
#include <algorithm>

typedef long double ldouble;
#define eps 1e-9

struct coodinate{
    ldouble x, y, z;
}cd[5];

struct sphere{
    coodinate c;
    ldouble r;
}sp[3];

inline ldouble f(ldouble x2, ldouble x1, ldouble k) {
    return (x2 * k * k - x1) / (k * k - 1);
}

inline ldouble f2(ldouble x2, ldouble x1, ldouble k) {
    return f(x2, x1, k) * f(x2, x1, k) - (k * k * x2 * x2 - x1 * x1) / (k * k - 1);
}

ldouble pi = acosl(-1);
ldouble dis(coodinate p, coodinate q) {
    ldouble ans = sqrt((p.x - q.x)*(p.x - q.x) + (p.y - q.y)*(p.y - q.y) + (p.z - q.z)*(p.z - q.z));
    return ans;
}

ldouble V(ldouble r) {
    return  (ldouble)4 / 3 * r * r * r * pi;//不加入强制转换结果会算错，详见下面的数据
}

int main() {
    int T;
    scanf("%d", &T);
    while (T--) {
        ldouble k1, k2, ans = 0;
        for (int i = 1; i <= 4; ++i)
            scanf("%Lf%Lf%Lf", &cd[i].x, &cd[i].y, &cd[i].z);
        scanf("%Lf%Lf", &k1, &k2);
        sp[1].c = coodinate{f(cd[2].x, cd[1].x, k1), f(cd[2].y, cd[1].y, k1), f(cd[2].z, cd[1].z, k1)};
        sp[1].r = sqrtl(f2(cd[2].x, cd[1].x, k1) + f2(cd[2].y, cd[1].y, k1) + f2(cd[2].z, cd[1].z, k1));
        sp[2].c = coodinate{f(cd[4].x, cd[3].x, k2), f(cd[4].y, cd[3].y, k2), f(cd[4].z, cd[3].z, k2)};
        sp[2].r = sqrtl(f2(cd[4].x, cd[3].x, k2) + f2(cd[4].y, cd[3].y, k2) + f2(cd[4].z, cd[3].z, k2));
        ldouble d = dis(sp[1].c, sp[2].c);
        ldouble maxr = std::max(sp[1].r, sp[2].r), minr = std::min(sp[2].r, sp[1].r);
        if (sp[1].r + sp[2].r <= d) ans = 0;
        else if (d + minr <= maxr) {
            ans = V(minr);
        } else {
            ldouble cosa = (sp[1].r * sp[1].r + d * d - sp[2].r * sp[2].r) / (2 * sp[1].r * d);
            ldouble cosb = (sp[2].r * sp[2].r + d * d - sp[1].r * sp[1].r) / (2 * sp[2].r * d);
            ans = pi / 3 * (3 * sp[1].r - sp[1].r * (1 - cosa)) * sp[1].r * (1 - cosa) * sp[1].r * (1 - cosa) + 
                    pi / 3 * (3 * sp[2].r - sp[2].r * (1 - cosb)) * sp[2].r * (1 - cosb) * sp[2].r * (1 - cosb);
        }
        printf("%Lf\n", ans);
    }
    return 0;
}
