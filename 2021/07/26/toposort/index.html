    拓扑排序

    1, 根据dfs的拓扑排序

    dfs可以找出天然的拓扑序，只是在最后需要倒转一下，想想为什么(因为最后一层先输出)

    遍历时，把遍历过程中的点的标记置为-1，如果再次访问到标记为-1的点，说明有环路，不能拓扑排序

    返回时，把遍历完的点置为1，表示这些点遍历完毕，没有环路，未来不用遍历了

    int dfs(int u, int n, int *ans) {
    flag[u] = -1;
    for (int v = 1; v <= n; ++v) {
        if (G[u][v]) {
            if (flag[v] == -1) return false;
            else if (flag[v] == 0 && !dfs(v, n, ans)) return false;
        }
    }
    flag[u] = 1;
    ans[cnt++] = u;
    return true;
}

int toposort(int n, int m, int *ans) {
    cnt = 0;
    memset(G, 0, sizeof(G));
    memset(flag, 0, sizeof(flag));
    for (int i = 0; i < m; ++i) {
        int x, y;
        scanf("%d%d", &x, &y);
        G[x][y] = 1;
    }
    for (int i = 1; i <= n; ++i) {
        if (!flag[i]) {
            if (!dfs(i, n, ans)) return false;
        }
    }
    reverse(ans, ans + cnt);
    return true;
}

    2, 根据bfs的拓扑排序

    原理为不停维护入度为0的点，让其入队，直到所有的点都被遍历到

    如果有环路，那么那个点不会入队，入度不可能被减为1，所以判断环路的条件就是是否遍历了所有点

    int G[MAXN][MAXN], in[MAXN];

int cnt = 0;
int toposort(int n, int m, int *ans) {
    std::queue<int> q;
    for (int i = 1; i <= n; ++i) {
        if (in[i] == 0) q.push(i);
    }
    while (!q.empty()) {
        int u = q.front();q.pop();
        ans[cnt++] = u;
        for (int v = 1; v <= n; ++v) {
            if (G[u][v]) {
                in[v]--;
                if (in[v] == 0) q.push(v);
            }
        }
    }
    return cnt == n;
}

int solve(int n, int m, int *ans) {
    memset(G, 0, sizeof(G));
    memset(in, 0, sizeof(in));
    cnt = 0;
    for (int i = 0; i < m; ++i) {
        int x, y;
        scanf("%d%d", &x, &y);
        G[x][y] = 1;
        in[y]++;
    }
    return toposort(n, m, ans);
}

    3, 根据层序的拓扑排序

    经典题目《车站分级》中的排序方式，可以找出拓扑序的层数

    int toposort(int G[][MAXN], int n, int *in, int *out) {
    int tmp[MAXN], book[MAXN];
    memset(book, 0, sizeof(book));
    int depth = 0;
    while (1) {
        int cnt = 0;
        for (int p = 1; p <= n; ++p) {
            if (!in[p] && !book[p]) {
                tmp[cnt++] = p;
                book[p] = 1;
            }
        }
        if (cnt == 0) break;
        for (int i = 0; i < cnt; ++i) {
            for (int j = 0; j < MAXN && out[tmp[i]] > 0; ++j) {
                if (G[tmp[i]][j]) {
                    in[j]--;
                    out[tmp[i]]--;
                }
            }
        }
        ++depth;
    }
    return depth;
}
