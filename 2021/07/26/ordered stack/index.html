    单调栈

    定义

    满足单调性的栈结构

    伪代码描述

    insert x
while !sta.empty() && sta.top()<x
    sta.pop()
sta.push(x)

    解决的问题

    定义函数 $f(i)$ 代表数列中第 $i$ 个元素之后第一个大于 $a_{i}$的元素的下标，即 $f(i)=min_{i<j\leq n, a_{j} > a_{i}}{j}$。若不存在，则 $f(i)=0$。
    求出$f(1) … f(n)$的值。

    样例输入

    5
    1 4 2 3 5

    样例输出

    2 5 4 5 0

    示例代码

    [USACO06NOV] POJ3250 Bad Hair Day

    #include <stdio.h>
#include <string.h>

#define MAXN 3000000 + 100

struct num {
    int p, val;
}stack[MAXN];

void Print_array(int *a, int n) {
    for (int i = 1; i <= n; ++i) {
        if (i == 1)printf("%d", a[i]);
        if (i > 1)printf(" %d", a[i]);
    }
    printf("\n");
}

int ans[MAXN];
int solve() {
    int top = 0;
    int n, x;
    scanf("%d", &n);
    for (int i = 1; i <= n; ++i) {
        scanf("%d", &x);
        struct num xx = {i, x};
        if (top == 0) stack[++top] = xx, ans[stack[top].p] = 0;
        else {
            while (top && xx.val > stack[top].val) {
                ans[stack[top].p] = xx.p;
                --top;
            }
            stack[++top] = xx;
        }
    }
    while (top--) {
        ans[stack[top].p] = 0;
    }
    Print_array(ans, n);
    return 1;
}
int main() {
    solve();
    return 0;
}
