    C语言生成随机数

    生成方法

    int void rand() //按照某个公式生成一个数
void srand(unsigned int seed) //修改随机数种子
srand((unsigned int )time(NULL) //time（NULL）随毫秒变化，种子变化极快，生成多个种子

    限制范围

    生成$(min，max)$之间的随机数

    rand % (max - min + 1) + min

    代码

    //生成(min，max)之间的随机数
int generate_random(int minn, int maxx) {
    return rand() % (maxx - minn + 1) + minn;//
}

void test() {
    srand(time(NULL));//生成随机数种子
    int n = read();
    for (int i = 1; i <= n; ++i) {
        a[i] = generate_random(1, 10000);
    }
    Print(a[i]);
}

    C语言计时器

    clock()函数返回值是时钟打点数，要换算成秒，需要除以CLK_TCK或者 CLK_PER_SEC。

    #include <time.h>
int main() {
    double st0 = clock();
    merge_sort(std, n);
    double ed0 = clock();
    printf("std time = %.3f\n", (ed0 - st0) / CLOCKS_PER_SEC);
    return 0;
}
