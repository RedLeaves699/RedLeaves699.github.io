    冒泡排序

    思想

    不停扫描数列，发现逆序就交换，直到没有逆序为止。最多扫描\(n\)次，至多进行\(n(n-1)/2\)次交换后，数列的顺序就会固定。因为一个长度为\(n\)的数列的逆序对最多就是\(n(n-1)/2\)个。由于交换数列中相邻的两个逆序数至多会让逆序对减少1，所以算法复杂度自然是\(O(n^2)\)的。

    图解

    实现

    #define T int
#define MAXN 1000 + 10
#define OP <

void bubbleSort(T *a, int n) {
    for (int i = 0; i < n; ++i) {
        for (int j = 0; j < n - i - 1; ++j) {
            if (a[j] < a[j + 1]);
            else {
                swap(&a[j + 1],&a[j]);
            }
        }
    }
}

void bubbleSort2(T *a, int n) {
    for (int i = 0; i < n; ++i) {
        for (int j = i + 1; j < n; ++j) {
            if (a[j] < a[i]) {
                swap(&a[i], &a[j]);
            }
        }
    }
}

    选择排序

    思想

    在未排序序列中找到最小（大）元素，存放到排序序列的起始位置。

    再从剩余未排序元素中继续寻找最小（大）元素，然后放到已排序序列的末尾。

    重复第二步，直到所有元素均排序完毕。

    图解

    实现

    void selectionSort(T *a, int n) {
    for (int i = 0; i < n; ++i) {
        int p = i;
        for (int j = i + 1; j < n; ++j) {
            if (a[j] < a[p]) {
                p = j;
            }
        }
        swap(&a[p], &a[i]);
    }
}

    插入排序

    思想

    将第一待排序序列第一个元素看做一个有序序列，把第二个元素到最后一个元素当成是未排序序列。

    从头到尾依次扫描未排序序列，将扫描到的每个元素插入有序序列的适当位置。（如果待插入的元素与有序序列中的某个元素相等，则将待插入元素插入到相等元素的后面。）

    好像我们在玩扑克牌时看牌，码牌的动作。

    图解

    实现

    void insertionSort(T *a,int n) {
    for (int i = 1; i < n; ++i) {
        int j = i - 1, t = a[i];
        for (; j >= 0 && t < a[j]; --j) {
            a[j + 1] = a[j];
        }
        a[j + 1] = t;
    }
}

    希尔排序

    思想

    对插入排序的一种优化。

    先将整个待排序的记录序列分割成为若干子序列分别进行直接插入排序，待整个序列中的记录"基本有序"时，再对全体记录进行依次直接插入排序。

    图解

    实现

    void shellSort(T *a,int n) {
    for (int gap = n >> 1; gap > 0; gap >>= 1) {
        for (int i = gap; i < n; ++i) {
            int j = i - gap, t = a[i];
            for (; j >= 0 && t < a[j]; j -= gap) {
                a[j + gap] = a[j];
            }
            a[j + gap] = t;
        }
    }
}

    快速排序

    可谓是无人不知无人不晓了，这里不多赘述。

    #define T int
#define OP <=
#define min(a, b)((a) < (b)? (a) : (b))

void swap(T *a, T *b) {
    T tmp = *b;
    *b = *a;
    *a = tmp;
}

void quick_sort(T *a, int start, int end) {
    if (start >= end) return;
    //int mid = start + (end - start) / 2;
    int i = start, j = end;
    register int num = a[start];
    //_swap(&a[start], &a[mid]);
    while (i < j) {
        while (a[j] >= num && i < j) j--;
        while (a[i] <= num && i < j) i++;
        if (i < j){
            swap(&a[i], &a[j]);
        }
    }
    swap(&a[start], &a[i]);
    quick_sort(a, start, i - 1);
    quick_sort(a, i + 1, end);
}

    归并排序

    非常常用的算法，是肯定要知道的，知名度仅次于快速排序。

    #define T int
#define OP <=
#define min(a, b)((a) < (b)? (a) : (b))

T temp[100000];
void merge(T *a, int start, int end1, int end2) {
    int i = start, j = end1 + 1, k = 0;
    while (i <= end1 && j <= end2) {
        if (a[i] OP a[j]) temp[k++] = a[i++];
        else temp[k++] = a[j++];
    }
    while (i <= end1) temp[k++] = a[i++];
    while (j <= end2) temp[k++] = a[j++];
    for (int i = 0; i <= end2 - start; ++i) {
        a[start + i] = temp[i];
    }
}

void merge_sort(T *a, int n) {
    for (int seg = 1; seg < n; seg = seg + seg) {
        for (int start = 0; start < n - seg; start += seg + seg) {
            merge(a, start, start + seg - 1, min(start + seg + seg - 1, n - 1));
        }
    }
}

    计数排序，桶排序

    计数排序

    统计每个元素出现了多少次，再根据统计结果输出，如图所示：

    T bucket[MAXN];
void countingSort(int *a, int n) {
    memset(bucket, 0, sizeof(bucket));
    for (int i = 1; i <= n; ++i) {
        assert(a[i] < MAXN && a[i] > 0);
        bucket[a[i]]++;
    }
    int k = 0;
    for (int i = 0; i < MAXN; ++i) {
        while (bucket[i]) {
            a[k++] = i;
            --bucket[i];
        }
    }
}

    桶排序

    计数排序的升级版，每个桶是一个区间，桶内数据先排序，然后按照计数排序的方法逐个从桶中取出数据。非常便于并行处理。

    基数排序

    思想

    是桶排序的变种，与桶排序按照区间分配数据不同，基数排序根据键值的每位数字来分配桶

    图解

    实现

    #define BASE 10
T b[MAXN];
void radixSort(T *a, int n) {  
    T m = a[0];
    int exp = 1;
    for (int i = 1; i < n; i++) {
        if (a[i] > m) {
            m = a[i];
        }
    }
    while (m / exp > 0) {
        int bucket[BASE] = { 0 };
        for (int i = 0; i < n; i++) {
            bucket[(a[i] / exp) % BASE]++;
        }
        for (int i = 1; i < BASE; i++) {
            bucket[i] += bucket[i - 1];
        }
        for (int i = n - 1; i >= 0; i--) {
            b[--bucket[(a[i] / exp) % BASE]] = a[i];
        }

        for (int i = 0; i < n; i++) {
            a[i] = b[i];
        }
        exp *= BASE;
    }
}

    堆排序

    使用二叉堆进行的排序，排序的思想就是二叉堆插入删除的原理。

    这里给出二叉堆的实现：

    #include <stdio.h>
#include <stdlib.h>
#include <time.h>
#include <math.h>

#define MAXN 1 << 16
#define OP >

void swap_c(int *a, int *b) {
    int t = *a;
    *a = *b;
    *b = t;


void heap_up(int *heap) {
    int size = heap[0];
    for (int i = size; i > 1 && heap[i] OP heap[i >> 1]; swap_c(&heap[i], &heap[i >> 1]), i >>= 1);
}
int son(int *heap, int x) { //find the son point to exchange
    int size = heap[0];
    return x * 2 + (x * 2 + 1 <= size && heap[x * 2 + 1] OP heap[x * 2]);
}
void heap_down(int *heap, int x) {
    int size = heap[0];
    for (int i = 1, t = son(heap, i); t <= size && heap[t] OP heap[i]; i = t, t = son(heap, i))
        swap_c(&heap[t], &heap[i]);
}
void heap_insert(int *heap, int x) {
    heap[heap[0]++] = x;
    heap_up(heap);
}
void heap_delete(int *heap) {
    int size = heap[0];
    swap_c(&heap[1], &heap[size]);
    heap[0]--;
    heap_down(heap, heap[1]);
}
void heap_build(int *heap, int size) {
    heap[0] = size;
    for (int i = size / 2; i > 0; --i) {
        heap_down(heap, i);
    }
}

int heap[MAXN];
void test() {
    srand(time(NULL));
    int n = (MAXN) / (1 << 10) - 1;n <<= 1;
    printf("%d\n", n);
    for (int i = 1; i < n; ++i) {
        heap[i] = generate_random(1, 10000);
    }
    heap_build(heap, n);
    Print_array(heap, n);
}
